คำสั่งควบคุมทิศทางการทำงาน


กรณีเงื่อนไขเดียว

คำสั่งควบคุมทิศทางการทำงาน ตัวอย่าง หลักการใช้
if if($a >= $b) { คำสั่ง } กรณีเงื่อนไขเดียว

คำสั่ง if ใช้เปรียบเทียบเงื่อนไขว่าเป็นจริงหรือเท็จ โดยมีการกำหนดคำสั่งที่ต้องการไว้ในบล็อกการทำงาน { }

<?php
$x = 20 ;

if ($x >= 10) {
print "$x >= 10" ;
}
?>

กรณีสองเงื่อนไข

คำสั่งควบคุมทิศทางการทำงาน ตัวอย่าง หลักการใช้
if else if($a >= $b) { คำสั่ง } else { คำสั่ง } กรณีสองเงื่อนไข

นำตัวแปรไปเปรียบเทียบในเงื่อนไขที่หนึ่งถ้าเป็นจริงจะทำตามคำสั่งในบล็อกของเงื่อนไขที่หนึ่ง if { } แต่ถ้าเป็นเท็จจะทำตามคำสั่งในบล็อกเงื่อนไขที่สอง else { }

<?php
$x = 20 ;

if ($x >= 15) {
print "$x >= 10" ;
}
else {
print "$x < 10" ;
}
?>

กรณีหลายเงื่อนไข

คำสั่งควบคุมทิศทางการทำงาน ตัวอย่าง หลักการใช้
if elseif else if($a >= $b){ คำสั่ง } elseif($a >= $b) { คำสั่ง } else { คำสั่ง } กรณีมากกว่าสองเงื่อนไข

เปรียบเทียบเงื่อนไขที่หนึ่งถ้าเป็นจริงจะทำตามคำสั่งในบล็อกของเงื่อนไขที่หนึ่ง if { } ถ้าเป็นเท็จจะไปเปรียบเทียบเงื่อนไขที่สอง
เปรียบเทียบเงื่อนไขที่สองถ้าเป็นจริงจะทำตามคำสั่งในบล็อกเงื่อนไขที่สอง else if { } แต่ถ้าเป็นเท็จจะทำตามคำสั่งในบล็อกเงื่อนไขที่สาม else { }

<?php
$x = 20 ;

if ($x >= 15) {
print "$x >= 15" ;
}
elseif ($x >= 10) {
print "$x >= 10" ;
}
else {
print "$x < 10" ;
}
?>

คำสั่ง switch

คำสั่งควบคุมทิศทางการทำงาน ตัวอย่าง หลักการใช้
switch switch($a) ใช้คำสั่ง switch เพื่อนำตัวแปรไปเปรียบเทียบในแต่ละ case
case case "red" ใช้คำสั่ง case เพื่อสร้างเงื่อนไขในการนำตัวแปรมาเปรียบเทียบ
break break ; คำสั่ง break เป็นการสั่งให้โปรแกรมออกจากคำสั่ง switch หากเจอเงื่อนไขแล้ว
default default : คำสั่ง default กำหนดการทำงาน ในกรณีไม่เข้าเงื่อนไขใด ๆ ของเงื่อนไขในแต่ละ case
<?php
$color = "red" ;

switch ($color) {
case "blue" ;
print "color is blue" ;
break ;

case "green" ;
print "color is green" ;
break ;

case "red" ;
print "color is red" ;
break ;

case "yellow" ;
print "color is yellow" ;
break ;

default :
print "no color" ;
}
?>
